# Rastgele Orman Sınıflandırıcı Örneği
from matplotlib import pyplot as plt
from sklearn.datasets import load_iris
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import plot_confusion_matrix
from sklearn.model_selection import train_test_split


def main():

    """
    sklearn fonksiyonunu kullanarak Rastgele Orman Sınıflandırıcı Örneği.
    Algoritmayı göstermek için Iris tipi veri seti kullanılmıştır.
    """

    # Iris veri setini yükle
    iris = load_iris()

    # Veri setini eğitim ve test verisi olarak ayır
    x = iris["data"]  # özellikler
    y = iris["target"]
    x_train, x_test, y_train, y_test = train_test_split(
        x, y, test_size=0.3, random_state=1
    )

    # Rastgele Orman Sınıflandırıcı
    rand_for = RandomForestClassifier(random_state=42, n_estimators=100)
    rand_for.fit(x_train, y_train)

    # Sınıflandırıcının Karışıklık Matrisi'ni Göster
    plot_confusion_matrix(
        rand_for,
        x_test,
        y_test,
        display_labels=iris["target_names"],
        cmap="Blues",
        normalize="true",
    )
    plt.title("Normalize Edilmiş Karışıklık Matrisi - IRIS Veri Seti")
    plt.show()


if __name__ == "__main__":
    main()
