# Rastgele Orman Regresörü Örneği
from sklearn.datasets import load_boston
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error, mean_squared_error
from sklearn.model_selection import train_test_split


def main():

    """
    sklearn fonksiyonunu kullanarak Rastgele Orman Regresörü Örneği.
    Algoritmayı göstermek için Boston ev fiyatları veri seti kullanılmıştır.
    """

    # Boston ev fiyatları veri setini yükle
    boston = load_boston()
    print(boston.keys())

    # Veri setini eğitim ve test verisi olarak ayır
    x = boston["data"]  # özellikler
    y = boston["target"]
    x_train, x_test, y_train, y_test = train_test_split(
        x, y, test_size=0.3, random_state=1
    )

    # Rastgele Orman Regresörü
    rand_for = RandomForestRegressor(random_state=42, n_estimators=300)
    rand_for.fit(x_train, y_train)

    # Test verileri için hedefi tahmin et
    tahminler = rand_for.predict(x_test)

    # Hata yazdırma
    print(f"Ortalama Mutlak Hata:\t {mean_absolute_error(y_test, tahminler)}")
    print(f"Ortalama Kare Hata  :\t {mean_squared_error(y_test, tahminler)}")


if __name__ == "__main__":
    main()
